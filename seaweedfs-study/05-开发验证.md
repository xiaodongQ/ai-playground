### 5.1.3 快速部署脚本

```bash
#!/bin/bash
# deploy-seaweedfs.sh - 一键部署测试集群

set -e

# 配置
DATA_DIR="/data/seaweedfs"
MASTER_HOST="10.0.1.1"
FILDER_HOST="10.0.1.1"
S3_HOST="10.0.1.1"

# 下载二进制
wget https://github.com/seaweedfs/seaweedfs/releases/latest/download/linux_amd64.tar.gz
tar -xzf linux_amd64.tar.gz
chmod +x weed

# 节点 1: Master + Filer + S3 + Volume
if [ "$(hostname)" == "node1" ]; then
    echo "Starting Master..."
    ./weed master -mdir=${DATA_DIR}/master -ip=${MASTER_HOST} -port=9333 &
    
    sleep 2
    
    echo "Starting Filer..."
    ./weed filer -master=${MASTER_HOST}:9333 -ip=${FILDER_HOST} -port=8888 &
    
    sleep 2
    
    echo "Starting S3 Gateway..."
    export AWS_ACCESS_KEY_ID=admin
    export AWS_SECRET_ACCESS_KEY=key
    ./weed s3 -filer=${FILDER_HOST}:8888 -ip=${S3_HOST} -port=8333 &
    
    echo "Starting Volume Server..."
    ./weed volume -dir=${DATA_DIR}/volume1 -master=${MASTER_HOST}:9333 -port=9340 &
fi

# 节点 2,3: Volume Server
if [ "$(hostname)" == "node2" ] || [ "$(hostname)" == "node3" ]; then
    echo "Starting Volume Server on $(hostname)..."
    ./weed volume -dir=${DATA_DIR}/volume1 -master=${MASTER_HOST}:9333 -port=9340 &
    ./weed volume -dir=${DATA_DIR}/volume2 -master=${MASTER_HOST}:9333 -port=9341 &
fi

echo "Deployment complete!"
```

### 5.2.2 文件上传下载测试

```python
# test-file-operations.py
import boto3
import hashlib
import os

# 配置
ENDPOINT = 'http://localhost:8333'
ACCESS_KEY = 'admin'
SECRET_KEY = 'key'
BUCKET = 'test-bucket'

def get_s3_client():
    return boto3.client(
        's3',
        endpoint_url=ENDPOINT,
        aws_access_key_id=ACCESS_KEY,
        aws_secret_access_key=SECRET_KEY,
        region_name='us-east-1'
    )

def test_upload_download():
    s3 = get_s3_client()
    
    # 创建 Bucket
    s3.create_bucket(Bucket=BUCKET)
    print(f"✓ Bucket '{BUCKET}' 创建成功")
    
    # 测试小文件
    small_file = 'test-small.txt'
    small_content = b'Hello, SeaweedFS!' * 100
    
    with open(small_file, 'wb') as f:
        f.write(small_content)
    
    # 上传
    s3.upload_file(small_file, BUCKET, 'small-file.txt')
    print(f"✓ 小文件上传成功 ({len(small_content)} bytes)")
    
    # 下载并验证
    s3.download_file(BUCKET, 'small-file.txt', 'downloaded-small.txt')
    with open('downloaded-small.txt', 'rb') as f:
        downloaded = f.read()
    
    assert downloaded == small_content, "文件内容不匹配!"
    print(f"✓ 小文件下载验证成功")
    
    # 测试大文件 (100MB)
    large_file = 'test-large.bin'
    large_size = 100 * 1024 * 1024  # 100MB
    
    with open(large_file, 'wb') as f:
        f.write(os.urandom(large_size))
    
    # 计算原始 MD5
    original_md5 = hashlib.md5(large_size.to_bytes(8, 'big') + open(large_file, 'rb').read()).hexdigest()
    
    # 上传
    s3.upload_file(large_file, BUCKET, 'large-file.bin')
    print(f"✓ 大文件上传成功 ({large_size / 1024 / 1024:.1f} MB)")
    
    # 下载并验证
    s3.download_file(BUCKET, 'large-file.bin', 'downloaded-large.bin')
    with open('downloaded-large.bin', 'rb') as f:
        downloaded_large = f.read()
    
    downloaded_md5 = hashlib.md5(large_size.to_bytes(8, 'big') + downloaded_large).hexdigest()
    assert original_md5 == downloaded_md5, "大文件内容不匹配!"
    print(f"✓ 大文件下载验证成功")
    
    # 清理
    s3.delete_object(Bucket=BUCKET, Key='small-file.txt')
    s3.delete_object(Bucket=BUCKET, Key='large-file.bin')
    s3.delete_bucket(Bucket=BUCKET)
    
    os.remove(small_file)
    os.remove('downloaded-small.txt')
    os.remove(large_file)
    os.remove('downloaded-large.txt')
    
    print("\n=== 所有测试通过! ===")

if __name__ == '__main__':
    test_upload_download()
```

import boto3
from boto3.s3.transfer import TransferConfig

ENDPOINT = 'http://localhost:8333'
ACCESS_KEY = 'admin'
SECRET_KEY = 'key'
BUCKET = 'test-multipart'

def test_multipart():
    s3 = boto3.client(
        's3',
        endpoint_url=ENDPOINT,
        aws_access_key_id=ACCESS_KEY,
        aws_secret_access_key=SECRET_KEY,
        region_name='us-east-1'
    )
    
    # 创建 Bucket
    s3.create_bucket(Bucket=BUCKET)
    
    # 创建 500MB 测试文件
    file_size = 500 * 1024 * 1024
    file_name = 'test-multipart.bin'
    
    print(f"创建 {file_size / 1024 / 1024:.0f}MB 测试文件...")
    with open(file_name, 'wb') as f:
        f.write(b'0' * file_size)
    
    # 配置分片上传
    config = TransferConfig(
        multipart_threshold=8 * 1024 * 1024,  # 8MB
        multipart_chunksize=8 * 1024 * 1024,  # 8MB
        max_concurrency=4,
        use_threads=True
    )
    
    # 分片上传
    print("开始分片上传...")
    s3.upload_file(file_name, BUCKET, 'multipart-test.bin', Config=config)
    print("✓ 分片上传成功")
    
    # 验证
    response = s3.head_object(Bucket=BUCKET, Key='multipart-test.bin')
    print(f"✓ 对象大小：{response['ContentLength'] / 1024 / 1024:.1f}MB")
    
    # 清理
    s3.delete_object(Bucket=BUCKET, Key='multipart-test.bin')
    s3.delete_bucket(Bucket=BUCKET)
    os.remove(file_name)
    
    print("=== 分片上传测试通过! ===")

if __name__ == '__main__':
    test_multipart()

import boto3

def test_bucket_operations():

### 5.4.3 大文件性能测试

```python
# benchmark-large-files.py
import boto3
import time
import os
from concurrent.futures import ThreadPoolExecutor

ENDPOINT = 'http://localhost:8333'
ACCESS_KEY = 'admin'
SECRET_KEY = 'key'
BUCKET = 'benchmark-large'

def get_s3_client():
    return boto3.client(
        's3',
        endpoint_url=ENDPOINT,
        aws_access_key_id=ACCESS_KEY,
        aws_secret_access_key=SECRET_KEY,
        region_name='us-east-1'
    )

def benchmark_single_thread(file_size_mb):
    """单线程大文件测试"""
    s3 = get_s3_client()
    s3.create_bucket(Bucket=BUCKET)
    
    file_name = f'test-{file_size_mb}MB.bin'
    key_name = f'{file_name}'
    
    # 创建文件
    print(f"创建 {file_size_mb}MB 测试文件...")
    with open(file_name, 'wb') as f:
        f.write(b'0' * file_size_mb * 1024 * 1024)
    
    # 上传
    print(f"上传 {file_size_mb}MB...")
    start = time.time()
    s3.upload_file(file_name, BUCKET, key_name)
    upload_time = time.time() - start
    upload_speed = file_size_mb / upload_time
    
    print(f"  上传时间：{upload_time:.2f}s")
    print(f"  上传速度：{upload_speed:.2f} MB/s")
    
    # 下载
    print(f"下载 {file_size_mb}MB...")
    start = time.time()
    s3.download_file(BUCKET, key_name, f'downloaded-{file_name}')
    download_time = time.time() - start
    download_speed = file_size_mb / download_time
    
    print(f"  下载时间：{download_time:.2f}s")
    print(f"  下载速度：{download_speed:.2f} MB/s")
    
    # 清理
    s3.delete_object(Bucket=BUCKET, Key=key_name)
    s3.delete_bucket(Bucket=BUCKET)
    os.remove(file_name)
    os.remove(f'downloaded-{file_name}')
    
    return upload_speed, download_speed

def benchmark_concurrent(file_size_mb, num_threads):
    """并发上传测试"""
    s3 = get_s3_client()
    s3.create_bucket(Bucket=BUCKET)
    
    def upload_file(thread_id):
        file_name = f'test-{thread_id}-{file_size_mb}MB.bin'
        with open(file_name, 'wb') as f:
            f.write(b'0' * file_size_mb * 1024 * 1024)
        
        start = time.time()
        s3.upload_file(file_name, BUCKET, f'{file_name}')
        elapsed = time.time() - start
        
        os.remove(file_name)
        return file_size_mb / elapsed
    
    print(f"并发上传测试 ({num_threads} 线程，每个 {file_size_mb}MB)...")
    start = time.time()
    
    with ThreadPoolExecutor(max_workers=num_threads) as executor:
        results = list(executor.map(upload_file, range(num_threads)))
    
    total_time = time.time() - start
    total_speed = sum(results)
    
    print(f"  总时间：{total_time:.2f}s")
    print(f"  总吞吐：{total_speed:.2f} MB/s")
    print(f"  平均速度：{total_speed/num_threads:.2f} MB/s/线程")
    
    # 清理
    for i in range(num_threads):
        s3.delete_object(Bucket=BUCKET, Key=f'test-{i}-{file_size_mb}MB.bin')
    s3.delete_bucket(Bucket=BUCKET)

if __name__ == '__main__':
    print("=== 单线程大文件测试 ===")
    for size in [10, 50, 100]:
        print(f"\n测试 {size}MB 文件:")
        benchmark_single_thread(size)
    
    print("\n=== 并发上传测试 ===")
    for threads in [1, 2, 4]:
        print(f"\n{threads} 线程并发:")
        benchmark_concurrent(10, threads)
```
