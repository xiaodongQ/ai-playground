### 5.2.2 第一个技能：计算器

**文件**: `~/.nanoclaw/skills/calculator.py`

```python
"""简单计算器技能"""

from __future__ import annotations

from nanoclaw.tools.registry import tool


@tool(
    name="calculate",
    description="执行数学计算。支持加减乘除和幂运算。",
    parameters={
        "expression": {
            "type": "string",
            "description": "数学表达式，例如：'2 + 2', '10 * 5', '2 ** 10'",
        }
    },
)
async def calculate(expression: str) -> str:
    """
    执行数学计算
    
    Args:
        expression: 数学表达式
    
    Returns:
        计算结果
    """
    try:
        # 安全检查：只允许数字和运算符
        allowed_chars = set("0123456789+-*/.() ")
        if not all(c in allowed_chars for c in expression):
            return "错误：表达式包含非法字符"
        
        # 计算结果
        result = eval(expression, {"__builtins__": {}}, {})
        return f"{expression} = {result}"
    
    except Exception as e:
        return f"计算错误：{e}"
```

"""GitHub Issue 查询技能"""

from __future__ import annotations

import aiohttp
from nanoclaw.tools.registry import tool


@tool(
    name="github_issue_search",
    description="搜索 GitHub 仓库的 Issues。返回匹配的 Issue 列表。",
    parameters={
        "owner": {
            "type": "string",
            "description": "仓库所有者 (用户名或组织名)",
        },
        "repo": {
            "type": "string",
            "description": "仓库名称",
        },
        "query": {
            "type": "string",
            "description": "搜索关键词",
        },
        "state": {
            "type": "string",
            "description": "Issue 状态：open, closed, all",
            "enum": ["open", "closed", "all"],
        },
    },
)
async def github_issue_search(
    owner: str,
    repo: str,
    query: str,
    state: str = "open",
) -> str:
    """
    搜索 GitHub Issues
    
    Args:
        owner: 仓库所有者
        repo: 仓库名称
        query: 搜索关键词
        state: Issue 状态
    
    Returns:
        搜索结果 (Markdown 格式)
    """
    from nanoclaw.core.llm import ConnectionPool
    
    # 构建搜索 URL
    search_url = f"https://api.github.com/search/issues"
    params = {
        "q": f"repo:{owner}/{repo} {query} state:{state}",
        "per_page": 5,
    }
    
    try:
        session = await ConnectionPool.get_session()
        
        async with session.get(search_url, params=params) as resp:
            if resp.status != 200:
                return f"GitHub API 错误：{resp.status}"
            
            data = await resp.json()
    
    except Exception as e:
        return f"搜索失败：{e}"
    
    # 格式化结果
    if data["total_count"] == 0:
        return f"未找到匹配的 Issues"
    
    results = [f"找到 {data['total_count']} 个 Issues:\n"]
    
    for item in data["items"]:
        results.append(
            f"## #{item['number']} {item['title']}\n"
            f"- 状态：{item['state']}\n"
            f"- 作者：{item['user']['login']}\n"
            f"- 创建时间：{item['created_at'][:10]}\n"
            f"- 链接：{item['html_url']}\n"
        )
    
    return "\n".join(results)
```

**文件**: `~/.nanoclaw/skills/weather.py`

```python
"""天气查询技能 (带缓存)"""

from __future__ import annotations

import time
from nanoclaw.tools.registry import tool


# 全局缓存
_weather_cache: dict[str, tuple[str, float]] = {}
CACHE_TTL = 600  # 10 分钟


@tool(
    name="get_weather_detailed",
    description="获取指定城市的详细天气信息。包括温度、湿度、风速等。",
    parameters={
        "city": {
            "type": "string",
            "description": "城市名称，例如：'Beijing', 'Shanghai', 'New York'",
        },
        "units": {
            "type": "string",
            "description": "温度单位：metric (摄氏度) 或 imperial (华氏度)",
            "enum": ["metric", "imperial"],
        },
    },
)
async def get_weather_detailed(city: str, units: str = "metric") -> str:
    """
    获取详细天气信息 (使用 Open-Meteo API，无需 API Key)
    
    Args:
        city: 城市名称
        units: 温度单位
    
    Returns:
        详细天气报告
    """
    from nanoclaw.core.llm import ConnectionPool
    
    # 检查缓存
    cache_key = f"{city}:{units}"
    if cache_key in _weather_cache:
        data, timestamp = _weather_cache[cache_key]
        if time.time() - timestamp < CACHE_TTL:
            return f"[从缓存] {data}"
    
    try:
        # 地理编码 (获取经纬度)
        session = await ConnectionPool.get_session()
        
        geo_url = f"https://geocoding-api.open-meteo.com/v1/search"
        geo_params = {"name": city, "count": 1, "language": "en", "format": "json"}
        
        async with session.get(geo_url, params=geo_params) as resp:
            if resp.status != 200:
                return f"地理编码失败：{resp.status}"
            
            geo_data = await resp.json()
            
            if not geo_data.get("results"):
                return f"未找到城市：{city}"
            
            location = geo_data["results"][0]
            lat = location["latitude"]
            lon = location["longitude"]
            city_name = location["name"]
            country = location.get("country", "")
        
        # 获取天气数据
        weather_url = f"https://api.open-meteo.com/v1/forecast"
        weather_params = {
            "latitude": lat,
            "longitude": lon,
            "current": "temperature_2m,relative_humidity_2m,wind_speed_10m,weather_code",
            "daily": "temperature_2m_max,temperature_2m_min,precipitation_sum",
            "timezone": "auto",
        }
        
        async with session.get(weather_url, params=weather_params) as resp:
            if resp.status != 200:
                return f"天气 API 失败：{resp.status}"
            
            weather_data = await resp.json()
        
        # 解析数据
        current = weather_data["current"]
        daily = weather_data["daily"]
        
        # 温度单位转换
        temp = current["temperature_2m"]
        if units == "imperial":
            temp = temp * 9/5 + 32
            temp_unit = "°F"
        else:
            temp_unit = "°C"
        
        # 天气代码解释
        weather_codes = {
            0: "晴朗",
            1: "主要晴朗",
            2: "多云",
            3: "阴天",
            45: "雾",
            48: "雾凇",
            51: "毛毛雨",
            61: "雨",
            71: "雪",
            95: "雷雨",
        }
        weather = weather_codes.get(current["weather_code"], "未知")
        
        # 格式化报告
        report = (
            f"## {city_name}, {country} 天气\n\n"
            f"### 当前天气\n"
            f"- 天气：{weather}\n"
            f"- 温度：{temp:.1f}{temp_unit}\n"
            f"- 湿度：{current['relative_humidity_2m']}%\n"
            f"- 风速：{current['wind_speed_10m']} km/h\n\n"
            f"### 今日预报\n"
            f"- 最高温度：{daily['temperature_2m_max'][0]}{temp_unit}\n"
            f"- 最低温度：{daily['temperature_2m_min'][0]}{temp_unit}\n"
            f"- 降水概率：{daily['precipitation_sum'][0]} mm\n"
        )
        
        # 保存到缓存
        _weather_cache[cache_key] = (report, time.time())
        
        return report
    
    except Exception as e:
        return f"天气查询失败：{e}"
```

**文件**: `nanoclaw/tools/database.py`

```python
"""数据库查询工具"""

from __future__ import annotations

import sqlite3
from pathlib import Path
from typing import Optional

from nanoclaw.core.logger import get_logger
from nanoclaw.tools.registry import tool

logger = get_logger(__name__)


@tool(
    name="database_query",
    description="在 SQLite 数据库中执行查询。仅支持 SELECT 语句 (安全考虑)。",
    parameters={
        "db_path": {
            "type": "string",
            "description": "数据库文件路径 (相对于工作区)",
        },
        "query": {
            "type": "string",
            "description": "SQL SELECT 查询语句",
        },
    },
    needs_confirmation=True,  # 需要用户确认
)
async def database_query(db_path: str, query: str) -> str:
    """
    执行数据库查询
    
    Args:
        db_path: 数据库路径
        query: SQL 查询
    
    Returns:
        查询结果 (表格格式)
    """
    from nanoclaw.security.sandbox import get_file_guard
    
    # 安全检查
    query_upper = query.strip().upper()
    if not query_upper.startswith("SELECT"):
        return "错误：仅支持 SELECT 查询"
    
    # 验证路径
    try:
        guard = get_file_guard()
        db_file = guard.validate_path(db_path)
        
        if not db_file.exists():
            return f"错误：数据库文件不存在：{db_path}"
    
    except Exception as e:
        return f"路径验证失败：{e}"
    
    # 执行查询
    try:
        conn = sqlite3.connect(str(db_file))
        cursor = conn.cursor()
        
        cursor.execute(query)
        rows = cursor.fetchall()
        
        if not rows:
            return "查询结果为空"
        
        # 获取列名
        columns = [desc[0] for desc in cursor.description]
        
        # 格式化为表格
        result = format_table(columns, rows)
        
        conn.close()
        return result
    
    except Exception as e:
        return f"查询失败：{e}"


def format_table(columns: list[str], rows: list[tuple]) -> str:
    """将查询结果格式化为表格"""
    if not rows:
        return "No results"
    
    # 计算列宽
    widths = [len(col) for col in columns]
    for row in rows:
        for i, val in enumerate(row):
            widths[i] = max(widths[i], len(str(val)))
    
    # 构建表格
    lines = []
    
    # 表头
    header = " | ".join(col.ljust(widths[i]) for i, col in enumerate(columns))
    lines.append(header)
    
    # 分隔线
    separator = "-+-".join("-" * w for w in widths)
    lines.append(separator)
    
    # 数据行
    for row in rows:
        line = " | ".join(str(val).ljust(widths[i]) for i, val in enumerate(row))
        lines.append(line)
    
    return "\n".join(lines)
```

### 5.4.2 添加新的安全层

**文件**: `nanoclaw/security/input_guard.py`

```python
"""输入验证安全层"""

from __future__ import annotations

import re
from typing import Optional

from nanoclaw.core.logger import get_logger

logger = get_logger(__name__)


class InputGuard:
    """用户输入验证"""
    
    # 最大输入长度
    MAX_INPUT_LENGTH = 10000
    
    # 禁止的命令模式
    FORBIDDEN_COMMANDS = [
        r"sudo\s",
        r"su\s",
        r"passwd",
        r"chmod\s+777",
    ]
    
    def __init__(self):
        self._compiled_patterns = [
            re.compile(p, re.IGNORECASE) for p in self.FORBIDDEN_COMMANDS
        ]
    
    def validate_input(self, user_input: str) -> tuple[bool, Optional[str]]:
        """
        验证用户输入
        
        Returns:
            (is_valid, error_message)
        """
        # 检查长度
        if len(user_input) > self.MAX_INPUT_LENGTH:
            return False, f"输入过长 (最大 {self.MAX_INPUT_LENGTH} 字符)"
        
        # 检查禁止的命令
        for pattern in self._compiled_patterns:
            if pattern.search(user_input):
                return False, f"检测到禁止的命令模式"
        
        return True, None


# 全局实例
_input_guard: Optional[InputGuard] = None


def get_input_guard() -> InputGuard:
    """获取全局 InputGuard 实例"""
    global _input_guard
    if _input_guard is None:
        _input_guard = InputGuard()
    return _input_guard
```

"""ChromaDB 语义搜索记忆存储"""

from __future__ import annotations

from typing import Optional

from nanoclaw.core.logger import get_logger

logger = get_logger(__name__)


class ChromaMemoryStore:
    """使用 ChromaDB 进行语义搜索"""
    
    def __init__(self, persist_directory: str):
        """
        初始化 ChromaDB 存储
        
        Args:
            persist_directory: 数据持久化目录
        """
        # 懒加载 chromadb (可选依赖)
        import chromadb
        
        self.client = chromadb.PersistentClient(path=persist_directory)
        self.collection = self.client.get_or_create_collection(
            name="memories",
            metadata={"hnsw:space": "cosine"}
        )
    
    async def add_memory(
        self,
        content: str,
        category: str = "general",
    ) -> str:
        """添加记忆"""
        import uuid
        
        memory_id = str(uuid.uuid4())
        
        self.collection.add(
            ids=[memory_id],
            documents=[content],
            metadatas=[{"category": category}],
        )
        
        return memory_id
    
    async def search_memories(
        self,
        query: str,
        limit: int = 5,
    ) -> list[dict]:
        """语义搜索记忆"""
        results = self.collection.query(
            query_texts=[query],
            n_results=limit,
            include=["documents", "metadatas", "distances"],
        )
        
        memories = []
        if results["documents"] and results["documents"][0]:
            for i, doc in enumerate(results["documents"][0]):
                memories.append({
                    "content": doc,
                    "category": results["metadatas"][0][i]["category"],
                    "distance": results["distances"][0][i],
                })
        
        return memories


# 工厂函数
def get_chroma_store() -> ChromaMemoryStore:
    """获取 ChromaDB 存储实例"""
    from pathlib import Path
    
    persist_dir = Path.home() / ".nanoclaw" / "chroma_data"
    persist_dir.mkdir(parents=True, exist_ok=True)
    
    return ChromaMemoryStore(str(persist_dir))

## 5.5 贡献代码

### 5.5.1 贡献流程

```
1. Fork 仓库
         │
2. 创建分支 (git checkout -b feature/my-feature)
         │
3. 开发功能 + 编写测试
         │
4. 运行测试 (pytest)
         │
5. 代码检查 (ruff + mypy)
         │
6. 提交代码 (git commit)
         │
7. 推送到 GitHub (git push)
         │
8. 创建 Pull Request
         │
9. 回应 Review 意见
         │
10. 合并到主分支
```

### 5.5.2 Git 工作流

```bash
# 1. Fork 仓库 (在 GitHub 上点击 Fork 按钮)

# 2. 克隆你的 Fork
git clone https://github.com/YOUR_USERNAME/nanoClaw.git
cd nanoClaw

# 3. 添加上游仓库
git remote add upstream https://github.com/ysz/nanoClaw.git

# 4. 创建功能分支
git checkout -b feature/add-new-skill

# 5. 开发和提交
# ... 编写代码 ...
git add skills/my_new_skill.py
git commit -m "feat: add my new skill for weather forecast"

# 6. 同步上游变更
git fetch upstream
git rebase upstream/main

# 7. 推送到你的 Fork
git push origin feature/add-new-skill

# 8. 在 GitHub 创建 Pull Request
```

### 5.5.3 提交信息规范

**格式**:
```
type: short description

Optional body explaining the change.

Fixes #123
```

**Type 类型**:
- `feat`: 新功能
- `fix`: Bug 修复
- `docs`: 文档更新
- `style`: 代码格式 (不影响功能)
- `refactor`: 重构 (不改变功能)
- `test`: 添加测试
- `chore`: 构建/工具配置

**示例**:
```
feat: add weather forecast skill

- Add get_weather_detailed skill with Open-Meteo API
- Support metric and imperial units
- Cache results for 10 minutes to reduce API calls

Fixes #45
```

```
fix: handle timeout in web_fetch tool

- Add 15 second timeout to web_fetch
- Return user-friendly error message on timeout
- Add test case for timeout scenario
```

### 5.5.4 Pull Request 指南

**PR 模板**:

```markdown
## Description
简要描述你的改动

## Type of Change
- [ ] Bug fix (non-breaking change which fixes an issue)
- [ ] New feature (non-breaking change which adds functionality)
- [ ] Breaking change (fix or feature that would cause existing functionality to not work as expected)
- [ ] Documentation update

## Testing
描述你如何测试这些改动

```bash
pytest tests/test_my_feature.py
```

## Checklist
- [ ] My code follows the style guidelines
- [ ] I have performed a self-review
- [ ] I have added tests that prove my fix/feature works
- [ ] All tests pass locally
- [ ] I have updated documentation accordingly
```

### 5.5.5 代码审查要点

**审查者会检查**:
1. **功能正确性**: 代码是否按预期工作
2. **测试覆盖**: 是否有足够的测试
3. **代码风格**: 是否符合项目规范
4. **安全性**: 是否有安全隐患
5. **性能影响**: 是否影响性能
6. **文档**: 是否更新了文档

**回应 Review**:
```bash
# 根据反馈修改代码
git add .
git commit -m "address review comments"

# 推送 (PR 会自动更新)
git push origin feature/my-feature
```
